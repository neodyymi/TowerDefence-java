115         root.getChildren().add(startButton);

143                 public void handle(long currentTime) {
144                     if (currentTime - previousTime < 100_000_000) {
145                         return;
146                     }
147                     playerStatus.set(game.getPlayer().toString());
148                     if (selectedTile != null) {
149                         tileStatus.set(selectedTile.toString());
150                     }
151 
152                     if (game.gameOver()) {
153                         Alert alert = new Alert(Alert.AlertType.INFORMATION);
154                         alert.setTitle("TowerDefence");
155                         alert.setHeaderText("Game Over - You Lost!");
156                         alert.setContentText("Restart game to start a new game.");
157                         alert.show();
158 
159                         this.stop();
160                     }
161 
162                     if (game.gameWon()) {
163                         Alert alert = new Alert(Alert.AlertType.INFORMATION);
164                         alert.setTitle("TowerDefence");
165                         alert.setHeaderText("Game Over - You Won!");
166                         alert.setContentText("Restart game to start a new game.");

168 
169                         this.stop();
170                     }

177                 Attacker[][] oldLocations = new Attacker[game.getMap().getHeight()][game.getMap().getWidth()];
178 
179                 @Override
180                 public void handle(long currentTime) {
181                     gameGrid.getChildren().stream().forEach((Node node) -> {

183                         Integer col = GridPane.getColumnIndex(node);
184                         Tile tile = game.getMap().getTile(GridPane.getRowIndex(node), GridPane.getColumnIndex(node));

190                         current.getChildren().add(createAttackerNode(tile.getAttackers()));

192                         if (t != null) {
193                             current.getChildren().add(createTowerNode(t, current));

195                     });
196 
197                 }

208     public Node createAttackerNode(List<Attacker> attackers) {
209 
210         FlowPane tile = new FlowPane(Orientation.VERTICAL);
211         tile.setAlignment(Pos.CENTER);
212         attackers.stream().forEach((Attacker a) -> {
213             Label attackerLabel = new Label("A");
214             Color color = Color.GREEN;
215             System.out.println("HEALTHPCT: " + a.getHealthPct());
216             if (a.getHealthPct() < 0.2) {
217                 color = Color.RED;
218             } else if (a.getHealthPct() < 0.4) {
219                 color = Color.ORANGE;
220             } else if (a.getHealthPct() < 0.6) {
221                 color = Color.YELLOW;
222             } else if (a.getHealthPct() < 0.8) {
223                 color = Color.YELLOWGREEN;
224             }
225             attackerLabel.setBackground(new Background(new BackgroundFill(color, CornerRadii.EMPTY, Insets.EMPTY)));
226             tile.getChildren().add(attackerLabel);
227         });
228         return tile;
229     }

231     public Node createTowerNode(Tower t, Pane node) {
232         StackPane towerImage = new StackPane();
233         Circle towerBase = new Circle();
234         towerBase.radiusProperty().bind(Bindings.min(towerImage.widthProperty().divide(2), towerImage.heightProperty().divide(2)));
235         Color color;
236         switch (t.getCharRepr()) {
237             case 'B':
238                 color = Color.CORNSILK;
239                 break;
240             case 'F':
241                 color = Color.AQUA;
242                 break;
243             default:
244                 color = Color.BLACK;
245         }
246         towerBase.setFill(color);
247         Text towerText = new Text(t.getCharRepr() + "");
248         towerImage.getChildren().addAll(towerBase, towerText);
249         return towerImage;
250     }

252     public void gameGridClick(int row, int col, Pane tile, MouseEvent e) {
253         System.out.println("Clicked " + row + ", " + col);
254         System.out.println("tile: " + tile);
255         if (tile.equals(this.selectedGridTile)) {
256             this.selectedGridTile = null;
257             this.selectedTile = null;
258             setSidepanelView(row, col, false);
259             return;
260         }
261         this.selectedGridTile = tile;
262         this.selectedTile = game.getMap().getTile(row, col);
263         this.selectedGridTile.setBorder(SELECTED_BORDER);
264 
265         setSidepanelView(row, col, true);
266     }

268     public VBox initSidePanel() {
269         VBox sidePanelDefaults = new VBox();
270         sidePanelDefaults.setAlignment(Pos.TOP_LEFT);
271         sidePanelDefaults.setPadding(new Insets(10, 0, 10, 0));
272         sidePanelDefaults.setBackground(new Background(new BackgroundFill(Color.CORAL, CornerRadii.EMPTY, Insets.EMPTY)));
273         sidePanelDefaults.setSpacing(10);
274         Text first = new Text();
275         Button startRoundButton = new Button("Start wave " + game.getNextWave());
276 
277         startRoundButton.setOnAction((ActionEvent e) -> {
278             game.startNextWave();
279             startRoundButton.setText("Start wave " + game.getNextWave());
280             startRoundButton.setDisable(true);
281             new AnimationTimer() {
282                 long previousTime = 0;
283 
284                 @Override
285                 public void handle(long currentTime) {
286                     if (currentTime - previousTime < TICK_NS) {
287                         return;
288                     }
289                     previousTime = currentTime;
290                     game.update();
291                     if (game.currentWaveFinished()) {
292                         this.stop();
293                         startRoundButton.setDisable(false);
294                     }
295                 }
296             }.start();
297         });
298 
299         first.textProperty().bind(this.playerStatus);
300         this.sidepanelContent = new VBox();
301         sidePanelDefaults.getChildren().addAll(first, sidepanelContent, startRoundButton);
302 
303         return sidePanelDefaults;
304     }

313     public Node specialViewFor(int row, int col) {
314         Tile mapTile = game.getMap().getTile(row, col);
315         VBox specialView = new VBox();
316         specialView.setAlignment(Pos.TOP_LEFT);
317         Text tileText = new Text();
318         tileText.textProperty().bind(tileStatus);
319         tileText.wrappingWidthProperty().bind(this.sidepanel.widthProperty());
320         if (mapTile.isBase()) {
321             specialView.getChildren().addAll(new Label("Base\n"), tileText);
322         } else if (mapTile.isBuildable()) {
323             if (mapTile.getTower() == null) {
324                 Button buyBasicTowerButton = new Button("Basic tower");
325                 buyBasicTowerButton.setOnAction((ActionEvent e) -> {
326                     Tower basicTower = new BasicTower(mapTile);
327                     if (!game.getPlayer().buy(basicTower)) {
328                         Alert alert = new Alert(Alert.AlertType.INFORMATION);
329                         alert.setTitle("TowerDefence");
330                         alert.setHeaderText("Not enough money.");
331                         alert.setContentText("You need more money to buy a basic tower!\n"
332                                 + "Upgrade cost: " + basicTower.getBuildCost() + "\n"
333                                 + "Available funds: " + game.getPlayer().getCurrency());
334                         alert.show();
335                     }
336                     setSidepanelView(row, col, true);
337                 });
338                 Button buyFreezeTowerButton = new Button("Freeze tower");
339                 buyFreezeTowerButton.setOnAction((ActionEvent e) -> {
340                     Tower freezeTower = new FreezeTower(mapTile);
341                     if (!game.getPlayer().buy(freezeTower)) {
342                         Alert alert = new Alert(Alert.AlertType.INFORMATION);
343                         alert.setTitle("TowerDefence");
344                         alert.setHeaderText("Not enough money.");
345                         alert.setContentText("You need more money to buy a freeze tower!\n"
346                                 + "Upgrade cost: " + freezeTower.getBuildCost() + "\n"
347                                 + "Available funds: " + game.getPlayer().getCurrency());
348                         alert.show();
349                     }
350                     setSidepanelView(row, col, true);
351                 });
352                 specialView.getChildren().addAll(new Label("Add tower"), buyBasicTowerButton, buyFreezeTowerButton);
353 
354             } else {
355                 Tower tower = mapTile.getTower();
356                 Button upgradeButton = new Button("Upgrade tower (" + tower.getUpgradeCost() + ")");
357                 upgradeButton.setOnAction((ActionEvent e) -> {
358                     if (!game.getPlayer().upgradeTower(tower)) {
359                         Alert alert = new Alert(Alert.AlertType.INFORMATION);
360                         alert.setTitle("TowerDefence");
361                         alert.setHeaderText("Not enough money.");
362                         alert.setContentText("You need more money to upgrade!\n"
363                                 + "Upgrade cost: " + tower.getUpgradeCost() + "\n"
364                                 + "Available funds: " + game.getPlayer().getCurrency());
365                         alert.show();
366                     }
367                     setSidepanelView(row, col, true);
368                 });
369                 Button sellButton = new Button("Sell tower (" + tower.getWorth() + ")");
370                 sellButton.setOnAction((ActionEvent e) -> {
371                     if (!game.getPlayer().sell(tower)) {
372                         Alert alert = new Alert(Alert.AlertType.INFORMATION);
373                         alert.setTitle("TowerDefence");
374                         alert.setHeaderText("Unable to sell");
375                         alert.setContentText("This shouldn't have happened...");
376                         alert.show();
377                     }
378                     setSidepanelView(row, col, true);
379                 });
380                 specialView.getChildren().addAll(new Label("Tower\n"), tileText, upgradeButton, sellButton);
381             }
382         } else if (mapTile.isSpawn()) {
383             specialView.getChildren().addAll(new Label("Spawn\n"), tileText);
384         } else {
385             specialView.getChildren().add(tileText);
386         }
387         return specialView;
388     }

390     public Scene createGameScene(Rectangle2D visualBounds) {
391         int mapWidth = this.game.getMap().getWidth();
392         int mapHeight = this.game.getMap().getHeight();
393 
394         int tileWidth = (int) ((visualBounds.getWidth() - SIDEBAR_WIDTH) / mapWidth);
395         int tileHeight = (int) (visualBounds.getHeight() / mapHeight);
396 
397         int width = tileWidth * mapWidth + SIDEBAR_WIDTH;
398         int height = tileHeight * mapHeight;
399 
400         BorderPane border = new BorderPane();
401 
402         this.gameGrid = new GridPane();
403         border.setCenter(this.gameGrid);
404         this.sidepanel = initSidePanel();
405         border.setRight(this.sidepanel);
406         this.sidepanel.setPrefWidth(SIDEBAR_WIDTH);
407         RowConstraints fullHeight = new RowConstraints();
408         fullHeight.setPercentHeight(100);
409 
410         for (int i = 0; i < mapHeight; i++) {
411             RowConstraints row = new RowConstraints();
412             row.setPercentHeight(100.0 / mapHeight);
413             this.gameGrid.getRowConstraints().add(row);
414             for (int j = 0; j < mapWidth; j++) {
415                 if (i == 0) {
416                     ColumnConstraints col = new ColumnConstraints();
417                     col.setPercentWidth(100.0 / (mapWidth));
418                     this.gameGrid.getColumnConstraints().add(col);
419                 }
420                 StackPane tile = new StackPane();
421                 tile.setAlignment(Pos.CENTER);
422 
423                 final int rowIndex = i;
424                 final int colIndex = j;
425                 tile.setOnMouseClicked((MouseEvent e) -> {
426                     gameGridClick(rowIndex, colIndex, tile, e);
427                 });
428                 tile.setBackground(this.game.getMap().getTile(i, j).getType().getBackground());
429                 tile.getChildren().add(new Label("" + i + "," + j));
430                 this.gameGrid.add(tile, j, i);
431 
432             }
433         }
434 
435         Scene gameScene = new Scene(border, width, height);
436 
437         return gameScene;
438     }
