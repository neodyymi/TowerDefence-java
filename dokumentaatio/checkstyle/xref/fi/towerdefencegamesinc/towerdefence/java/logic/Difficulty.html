6   package fi.towerdefencegamesinc.towerdefence.java.logic;
7   
8   /**
9    * Enumerator for dealing with game difficulty. It holds some settings 
10   * for the difficulty level
11   * @author vrsaari
12   */
13  public enum Difficulty {
14      EASY("Easy", 0),
15      NORMAL("Normal", 1),
16      HARD("Hard", 2);
17  
18      private final String name;
19      private final int difficulty;
20  
21      private Difficulty(String name, int difficulty) {
22          this.name = name;
23          this.difficulty = difficulty;
24      }
25  
26      public int getDifficulty() {
27          return difficulty;
28      }
29  
30      public String getName() {
31          return name;
32      }
33  
34      @Override
35      public String toString() {
36          return this.name;
37      }
38  
39  }
