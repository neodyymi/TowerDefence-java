6   package fi.towerdefencegamesinc.towerdefence.java.logic;
7   
8   /**
9    * Class for handling locations or coordinates.
10   * @author vrsaari
11   */
12  public class Location {
13  
14      private final int x;
15      private final int y;
16  
17      /**
18       * Creates a new location.
19       * @param x x-coordinate for location.
20       * @param y y-coordinate for location.
21       */
22      public Location(int x, int y) {
23          this.x = x;
24          this.y = y;
25      }
26  
27      /**
28       * Calculate distance between two locations.
29       * @param x1 x-coordinate for first tile.
30       * @param y1 y-coordinate for first tile.
31       * @param x2 x-coordinate for second tile.
32       * @param y2 y-coordinate for second tile.
33       * @return distance between coordinates.
34       */
35      public static double getDistance(int x1, int y1, int x2, int y2) {
36          int deltaX = Math.abs(x1 - x2);
37          int deltaY = Math.abs(y1 - y2);
38          double distance = Math.sqrt(deltaX * deltaX + deltaY * deltaY);
39          return distance;
40      }
41  
42      /**
43       * Calculate distance between two locations.
44       * @param location First location.
45       * @param other Other location.
46       * @return distance between locations.
47       */
48      public static double getDistance(Location location, Location other) {
49          return Location.getDistance(location.x, location.y, other.x, other.y);
50      }
51  
52      @Override
53      public String toString() {
54          return ("(" + this.x + ", " + this.y + ")");
55      }
56      
57      
58  
59  }
