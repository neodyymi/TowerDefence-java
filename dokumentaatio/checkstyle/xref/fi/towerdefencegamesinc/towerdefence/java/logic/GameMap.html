63      /**
64       * Read information from a file to generate a map.
65       *
66       * @param fileName File to read map information from.
67       * @param external Is the map file an external file?
68       * @return Generated map object.
69       */
70      public static GameMap loadMapFromFile(String fileName, boolean external) {
71          char[][] tmpTiles;
72          if (external) {
73              try (Stream<String> stream = Files.lines(Paths.get(MAPS_PATH + fileName))) {
74                  tmpTiles = stream.map(s -> s.toCharArray()).toArray(char[][]::new);
75              } catch (IOException e) {
76                  e.printStackTrace(System.out);
77                  return null;
78              }
79          } else {
80              try (InputStream is = GameMap.class.getClassLoader().getResourceAsStream(MAPS_PATH + fileName);
81                      final Reader r = new InputStreamReader(is, StandardCharsets.UTF_8);
82                      final BufferedReader br = new BufferedReader(r);
83                      final Stream<String> stream = br.lines()) {
84                  tmpTiles = stream.map(s -> s.toCharArray()).toArray(char[][]::new);
85                  is.close();
86              } catch (IOException e) {
87                  e.printStackTrace(System.out);
88                  return null;
89              }
90          }
91          GameMap map = loadMapFromCharArray(tmpTiles);
92  
93          return map;
94      }

96      /**
97       * Mostly a assisting method for generating a map from a file.
98       *
99       * @param tmpTiles 2d char-array used to generate a map from.
100      * @return Generated map object.
101      */
102     public static GameMap loadMapFromCharArray(char[][] tmpTiles) {
103         int width = tmpTiles[0].length;
104         int height = tmpTiles.length;
105         GameMap map = new GameMap(height, width);
106         for (int i = 0; i < height; i++) {
107             for (int j = 0; j < width; j++) {
108                 switch (tmpTiles[i][j]) {
109                     case ' ':
110                         map.tiles[i][j] = new Tile(j, i, Type.Road, false);
111                         break;
112                     case '#':
113                         map.tiles[i][j] = new Tile(j, i, Type.Unbuildable, false);
114                         break;
115                     case '@':
116                         map.tiles[i][j] = new Tile(j, i, Type.Buildable, true);
117                         break;
118                     case 'X':
119                         map.tiles[i][j] = new Tile(j, i, Type.Spawn, false);
120                         map.addSpawn(map.tiles[i][j]);
121                         break;
122                     case '$':
123                         map.tiles[i][j] = new Tile(j, i, Type.Base, false);
124                         break;
125                     default:
126                         break;
127                 }
128 
129             }
130         }
131         for (int i = 0; i < height; i++) {
132             for (int j = 0; j < width; j++) {
133                 if (i > 0) {
134                     map.tiles[i][j].setNorth(map.tiles[i - 1][j]);
135                 }
136                 if (i < map.tiles.length - 1) {
137                     map.tiles[i][j].setSouth(map.tiles[i + 1][j]);
138                 }
139                 if (j < map.tiles[i].length - 1) {
140                     map.tiles[i][j].setEast(map.tiles[i][j + 1]);
141                 }
142                 if (j > 0) {
143                     map.tiles[i][j].setWest(map.tiles[i][j - 1]);
144                 }
145             }
146         }
147         return map;
148     }

150     @Override
151     public String toString() {
152         StringBuilder sb = new StringBuilder();
153         StringBuilder attackerTiles = new StringBuilder();
154         Stream.of(tiles).forEach(x -> {
155             Stream.of(x).forEach(tile -> {
156                 Tile t = (Tile) tile;
157                 if (t.getTower() == null && t.getAttackers().isEmpty()) {
158                     if (null != t.getType()) {
159                         switch (t.getType()) {
160                             case Road:
161                                 sb.append(Type.Road.toString());
162                                 break;
163                             case Unbuildable:
164                                 sb.append(Type.Unbuildable.toString());
165                                 break;
166                             case Buildable:
167                                 sb.append(Type.Buildable.toString());
168                                 break;
169                             case Spawn:
170                                 sb.append(Type.Spawn.toString());
171                                 break;
172                             case Base:
173                                 sb.append(Type.Base.toString());
174                             default:
175                                 break;
176                         }
177                     }
178                 } else if (t.getTower() != null) {
179                     sb.append(t.getTower().getCharRepr());
180                 } else {
181                     int count = t.getAttackers().size();
182                     if (count < 10) {
183                         sb.append(count);
184                     } else {
185                         sb.append("*");
186                     }
187                     attackerTiles.append(t.toString()).append("\n");
188                 }
189             });
190             sb.append("\n");
191         });
192         sb.append("\n").append(attackerTiles);
193         return sb.toString();
194     }

275         Arrays.stream(this.tiles).forEach(row -> Arrays.stream(row).forEach(t -> {

322     /**
323      * What external map files are available in the /maps/ folder.
324      *
325      * @return List of external maps available.
326      * @throws IOException
327      */
328     public static List<String> externalMapFiles() throws IOException {
329         if (!Files.isDirectory(Paths.get(MAPS_PATH))) {
330             return new ArrayList<>();
331         }
332         File dir = new File(MAPS_PATH);
333         if (!dir.exists() || !dir.isDirectory()) {
334             return null;
335         }
336         ArrayList<String> files = new ArrayList(Arrays.asList(dir.list((d, name) -> name.endsWith(".map"))));
337 
338         return files.stream().map(m -> EXTERNAL_MAP_PREFIX + m).collect(Collectors.toList());
339     }
