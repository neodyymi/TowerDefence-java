110     /**
111      * Attempt to add a tower on the tile.
112      *
113      * @param tower The tower to be added to the tile.
114      * @return Was the tower added successfully?
115      */
116     public boolean addTower(Tower tower) {
117         if (this.tower == null && this.buildable) {
118             this.tower = tower;
119             return true;
120         } else {
121             return false;
122         }

131     public boolean removeTower(Tower tower) {
132         if (this.tower.equals(tower)) {
133             this.tower = null;
134             return true;
135         } else {
136             return false;

164     @Override
165     public String toString() {
166         StringBuilder sb = new StringBuilder();
167         sb.append(this.type.toString());
168         sb.append(" ");
169         sb.append(this.location);
170         sb.append(" ");
171         if (buildable) {
172             sb.append("Buildable\n");
173         } else {
174             sb.append("Unbuildable\n");
175         }
176 
177         if (this.tower != null) {
178             sb.append(this.tower.toString());
179         }
180         if (!this.attackers.isEmpty()) {
181             this.attackers.stream().forEach(x -> sb.append("\n").append(x.toString()));
182         }
183 
184         return sb.toString();
185     }

196     /**
197      * Find the next road tile to move to.
198      *
199      * @param previous The tile visited before this one.
200      * @return The next tile to move to.
201      */
202     public Tile nextRoad(Tile previous) {
203         Tile next = null;
204         List<Tile> options = new ArrayList();
205         if (this.east != null) {
206             options.add(this.east);
207             //System.out.println("East: " + this.east.toString());
208         }
209         if (this.south != null) {
210             options.add(this.south);
211             //System.out.println("South: " + this.south.toString());
212         }
213         if (this.north != null) {
214             options.add(this.north);
215             //System.out.println("North: " + this.north.toString());
216         }
217         if (this.west != null) {
218             options.add(this.west);
219             //System.out.println("West: " + this.west.toString());
220         }
221         Collections.shuffle(options);
222         while (next == null) {
223             if (options.isEmpty()) {
224                 return null;
225             }
226             next = options.remove(0);
227 
228             if (!next.canMoveTo()) {
229                 next = null;
230             } else if (next == previous) {
231                 next = null;
232             } else {
233                 break;
234             }
235         }
236         return next;
237     }
238 
239     private boolean canMoveTo() {
240         return this.type == Type.Road || this.type == Type.Spawn || this.type == Type.Base;
241     }
