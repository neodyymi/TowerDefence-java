6   package fi.towerdefencegamesinc.towerdefence.java.logic.tower;
7   
8   import fi.towerdefencegamesinc.towerdefence.java.logic.Tile;
9   import fi.towerdefencegamesinc.towerdefence.java.logic.attacker.Attacker;
10  
11  /**
12   * A tower with capability to freeze or slow enemies.
13   *
14   * @author vrsaari
15   */
16  public class FreezeTower extends BasicTower {
17  
18      private static final int COST = 200;
19      private static final int[] UPGRADE_COST = new int[]{200, 400, 600, 800, 1000};
20  
21      /**
22       * Create a freeze tower with the given parameters.
23       *
24       * @param tile The tile the tower is located in.
25       * @param power The power of the tower.
26       * @param speed The firingspeed of the tower.
27       * @param range The range of the tower.
28       * @param upgradeCost Array of costs to upgrade the tower.
29       */
30      public FreezeTower(Tile tile, int power, int speed, double range, int[] upgradeCost) {
31          super(tile, power, speed, range, upgradeCost, COST);
32      }
33  
34      /**
35       * Create a freeze tower with set parameters.
36       *
37       * @param tile The tile the tower is located in.
38       */
39      public FreezeTower(Tile tile) {
40          this(tile, 1, 1000, 2, UPGRADE_COST);
41      }
42  
43      /**
44       * Create a freeze tower with set parameters.
45       */
46      public FreezeTower() {
47          this(null);
48      }
49  
50      @Override
51      public int shoot(Attacker target) {
52          return super.shoot(target);
53      }
54  
55      @Override
56      public char getCharRepr() {
57          return 'F';
58      }
59  
60  }
